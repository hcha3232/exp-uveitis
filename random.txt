this.container.querySelectorAll('.form-check-input').forEach(input => {
            input.addEventListener('click', (event) => {
                this.tempSelectedOption = event.target.value;

                this.container.querySelectorAll('.form-check').forEach(input => {
                    input.classList.remove('border-success');
                });
                // Add 'border-success' class to the selected element
                event.target.closest('.form-check').classList.add('border-success');   
            });
        });
